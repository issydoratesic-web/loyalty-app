    <script>
        const TOTAL_STAMPS = 10;
        
        // Tajna šifra koju mora sadržavati tvoj QR kod
        const SALON_SECRET_CODE = "ADORA-TAJNA-LOZINKA-2026";
        
        let currentStamps = parseInt(localStorage.getItem('adora_stamps')) || 0;
        
        const stampsBox = document.getElementById('stamps-box');
        const scanBtn = document.getElementById('scan-btn');
        const claimBtn = document.getElementById('claim-btn');
        const scannerSection = document.getElementById('scanner-section');
        const closeScannerBtn = document.getElementById('close-scanner-btn');
        
        let html5QrcodeScanner = null;

        // Generiranje kartica/štambilja
        function renderStamps() {
            stampsBox.innerHTML = '';
            for (let i = 1; i <= TOTAL_STAMPS; i++) {
                const stampEl = document.createElement('div');
                stampEl.classList.add('stamp');
                
                if (i === 5) {
                    stampEl.classList.add('special');
                    stampEl.innerHTML = i <= currentStamps ? '🌸' : '-50%';
                } else if (i === 10) {
                    stampEl.classList.add('special');
                    stampEl.innerHTML = i <= currentStamps ? '🎁' : 'GRATIS';
                } else {
                    stampEl.innerHTML = i <= currentStamps ? '✨' : i;
                }

                if (i <= currentStamps) {
                    stampEl.classList.add('active');
                }

                stampsBox.appendChild(stampEl);
            }
            
            checkRewards();
        }

        // Provjera jesu li ostvarene nagrade i prikaz gumba za preuzimanje
        function checkRewards() {
            if (currentStamps === 5) {
                claimBtn.style.display = 'block';
                claimBtn.innerText = 'Iskoristi 50% popusta';
                scanBtn.style.display = 'none';
            } else if (currentStamps === 10) {
                claimBtn.style.display = 'block';
                claimBtn.innerText = 'Preuzmi BESPLATAN termin';
                scanBtn.style.display = 'none';
            } else {
                claimBtn.style.display = 'none';
                scanBtn.style.display = 'block';
            }
        }

        // Akcija za preuzimanje nagrade
        claimBtn.addEventListener('click', () => {
            if (currentStamps === 5) {
                alert("Čestitamo! Ostvarili ste 50% popusta na ovaj tretman. Pokažite ovaj ekran osoblju.");
                currentStamps = 6; // Prebaci na sljedeći korak da nastavi skupljati do 10
            } else if (currentStamps === 10) {
                alert("Čestitamo! Ostvarili ste BESPLATAN termin. Hvala vam na vjernosti! Kartica se sada resetira.");
                currentStamps = 0; // Resetira se na nulu
            }
            localStorage.setItem('adora_stamps', currentStamps);
            renderStamps();
        });

        // Pokretanje kamere klijenta za skeniranje tvog QR koda
        scanBtn.addEventListener('click', () => {
            scannerSection.style.display = 'flex';
            
            html5QrcodeScanner = new Html5QrcodeScanner(
                "reader", { fps: 10, qrbox: 250 }, false
            );
            
            html5QrcodeScanner.render(onScanSuccess, onScanError);
        });

        // Gašenje kamere
        closeScannerBtn.addEventListener('click', closeScanner);

        function closeScanner() {
            if (html5QrcodeScanner) {
                html5QrcodeScanner.clear().then(() => {
                    scannerSection.style.display = 'none';
                }).catch(err => {
                    console.error("Greška pri gašenju kamere: ", err);
                    scannerSection.style.display = 'none';
                });
            } else {
                scannerSection.style.display = 'none';
            }
        }

        // Što se događa kad klijent uspješno skenira TVOJ kod
        function onScanSuccess(decodedText, decodedResult) {
            closeScanner();
            
            // Provjera je li skeniran točan kod tvog salona
            if (decodedText === SALON_SECRET_CODE) {
                if (currentStamps < TOTAL_STAMPS) {
                    currentStamps++;
                    localStorage.setItem('adora_stamps', currentStamps);
                    renderStamps();
                    alert("Uspješno skenirano! Dodan Vam je novi štambilj u salonu ADORA. ✨");
                } else {
                    alert("Vaša kartica je već puna! Iskoristite besplatan termin.");
                }
            } else {
                alert("Nevažeći kod salona. Molimo skenirajte službeni ADORA QR kod.");
            }
        }

        function onScanError(err) {
            // Ignorira se kako se ne bi stalno pojavljivali sitni logovi grešaka tijekom traženja fokusa
        }

        // Prvo učitavanje kartice
        renderStamps();
    </script>
